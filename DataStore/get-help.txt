Add-Dev 						add a developer to the table
	-n <required> 	 				"name" parameter
	-ln <required> 	 				"last name" parameter
	-ph <optional def: "-"> 	 		"phone" paramert
	-phtype <optional def: "mobile"> 		"phone type" parameter
	-o <optional; def: 0; restrict: int >= 0>	"office" parameter, by office's "id"
	-plang <optional def: "unknown"> 		"programming language" parameter
	-lvl <optional def: "not assigned">		"professional level" parameter
